<script>
function toggleChat() {
  let box = document.getElementById("chatBox");
  box.style.display = box.style.display === "flex" ? "none" : "flex";
}

function sendMessage() {
  let input = document.getElementById("userInput");
  let text = input.value;
  if (!text) return;

  let chat = document.getElementById("chatBody");

  chat.innerHTML += "<p><b>Ти:</b> " + text + "</p>";

  let answer = "Я поки вчуся 🙂";

  if (text.toLowerCase().includes("план")) {
    answer = "План формується у SharePoint у відповідному списку.";
  }

  if (text.toLowerCase().includes("строк")) {
    answer = "Строки визначаються відповідно до плану заходів.";
  }

  chat.innerHTML += "<p><b>Бот:</b> " + answer + "</p>";

  input.value = "";
  chat.scrollTop = chat.scrollHeight;
}
</script>
